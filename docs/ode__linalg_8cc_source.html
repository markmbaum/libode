25 void ode_crout_LU (double **A, int n, int *p) {
26 
27  int i, j, k, idx, ti;
28  double m, td;
29 
30  //initialize the permutation array
31  for (i=0; i<n; i++) p[i] = i;
32 
33  //move through the matrix
34  for (i=0; i<n; i++) {
35  //find the largest element in the column beneath A[i][i]
36  m = 0.0;
37  idx = i;
38  for (j=i; j<n; j++) {
39  td = fabs(A[j][i]);
40  if ( td > m ) {
41  m = td;
42  idx = j;
43  }
44  }
45  //if all elements are zero, the matrix is singular
46  if ( m == 0.0 ) {
47  printf("\nLINALG FAILURE: the matrix is singular\n\n");
48  exit(EXIT_FAILURE);
49  }
50  //if the largest element isn't on the diagonal, pivot
51  if ( idx != i ) {
52  //swap rows
53  for (j=0; j<n; j++) {
54  td = A[i][j];
55  A[i][j] = A[idx][j];
56  A[idx][j] = td;
57  }
58  //swap permutation indices
59  ti = p[i];
60  p[i] = p[idx];
61  p[idx] = ti;
62  }
63  //continue with the decomposition
64  for (j=i+1; j<n; j++)
65  A[j][i] /= A[i][i];
66  for (j=i+1; j<n; j++)
67  for (k=i+1; k<n; k++)
68  A[j][k] -= A[j][i]*A[i][k];
69  }
70 }
